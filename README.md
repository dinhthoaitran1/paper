=[SYNTAX]=>

cd paper-main

+ Check source code:
=[SYNTAX]=>
cat paper.pl | less

+ Check demo template:
=[SYNTAX]=>
cat ppr/tpl_resume.html | less

+ Check demo template:
=[SYNTAX]=>
cat ppr/dat_resume.txt | less

+ Run Perl 5 script:
=[SYNTAX]=>
perl paper.pl ppr/dat_resume.txt ppr/tpl_resume.html ppr/rsl_resume.html

+ Check result HTML file:
=[SYNTAX]=>
cat ppr/rsl_resume.html | less
